Under the Hood

The logic, exposed.

This is the actual lead-routing function from the CRM automation project. No off-the-shelf plugins — every rule is explicit, testable, and owned by the client.

lead_router.py 
# lead_router.py — JO4 Dev CRM Automation Engine
# Distributes inbound leads to agents based on area + budget rules.

from dataclasses import dataclass
from typing import Optional
from .models import Agent, Lead, Assignment
from .notifications import send_agent_alert


@dataclass
class RoutingResult:
    agent: Optional[Agent]
    reason: str
    confidence: float  # 0.0 – 1.0


def route_lead(lead: Lead, agents: list[Agent]) -> RoutingResult:
    """
    Route an inbound lead to the best available agent.

    Priority:
        1. Area match (suburb in agent's coverage zones)
        2. Budget within agent's mandate bracket
        3. Least-loaded agent among qualifying candidates
    """
    candidates = [
        a for a in agents
        if a.is_active and lead.suburb in a.coverage_zones
    ]

    if not candidates:
        return RoutingResult(
            agent=None,
            reason="no_area_match",
            confidence=0.0
        )

    # Filter by budget bracket
    budget_match = [
        a for a in candidates
        if a.min_mandate <= lead.budget <= a.max_mandate
    ]
    pool = budget_match or candidates  # fall back to area-only match
    confidence = 0.95 if budget_match else 0.65

    # Assign to least-loaded agent
    chosen = min(pool, key=lambda a: a.open_leads_count)

    assignment = Assignment.create(lead=lead, agent=chosen)
    assignment.save()
    send_agent_alert(chosen, lead)

    return RoutingResult(
        agent=chosen,
        reason="area_and_budget_match" if budget_match else "area_only_match",
        confidence=confidence
    )
